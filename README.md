char LittleStar[42] = {
	'c','c','G','G','A','A','G',
	'F','F','E','E','D','D','c',
	'G','G','F','F','E','E','D',
	'c','c','G','G','A','A','G',
	'F','F','E','E','D','D','c',
};

int main(int argc, char* argv[])
{
	cout << "Play music !";
	_getch();
	char Tone = 0;


	for (int i = 0; i < 42; i++)
	{
		Tone = LittleStar[i];

		switch (Tone)
		{
		case 'c':
			Beep(523.25, 500);
			break;
